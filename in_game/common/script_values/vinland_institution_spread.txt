# Institution spread values adjusted for Vinland mod (we dont get overland spread, only trade spread)

	value = development
	multiply = 0.03
	add = {
		value = average_location_literacy
		multiply = 0.1
	}
	if = {
		limit = {
			OR = {
            	owner.culture = culture:icelandic

        	}
		}
